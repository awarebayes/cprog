float vec_len(float a_x, float a_y);
float vec_dot(float a_x, float a_y, float b_x, float b_y,
float c_x, float c_y);

int main()
{
    ///< Points
    float ax, ay; 
    float bx, by;
    float cx, cy;
    ///< Angles
    float dot_a, dot_b, dot_c;
    float min_dot;

    if (scanf("%f %f", &ax, &ay) != 2 ||
        scanf("%f %f", &bx, &by) != 2 ||
        scanf("%f %f", &cx, &cy) != 2)
    {
        printf("Input Error\n");
        return 1;
    }

    dot_a = vec_dot(bx, by, ax, ay, cx, cy);
    dot_b = vec_dot(ax, ay, bx, by, cx, cy);

/*
 * min of 2 integers
 */
float min(float a, float b)
{
    if (a > b)
        return b;
    return a;
}

/*
 * is value x close to target
 */
bool is_close(const float x, const float target)
{
    return fabs(x - target) < 10e-5;
}

/*
 * Sign of Float
 * if close to 0 = 0
 * else = 
 */
int sign(float f)
{
    if(is_close(f, 0))
        return 0;
    return (f > 0) ? 1 : -1;
}

/*
 * sign of cosine between two vectors of points
 * @param *_* - vector's x/y component
 * @return Angle between two vectors: ab and bc
 */

float vec_dot(float a_x, float a_y, float b_x,
float b_y, float c_x, float c_y)
{
    float ba_x, ba_y, bc_x, bc_y; 
    float dot; ///< dot product
    ba_x = a_x - b_x;
    ba_y = a_y - b_y; 
    bc_x = c_x - b_x;
    bc_y = c_y - b_y;

    dot = (ba_x * bc_x) + (ba_y * bc_y); 
    if (is_close(dot, 0))
        return 0;
    return dot;
}
